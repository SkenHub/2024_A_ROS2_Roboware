        function Timer_button() {
            if (count % 2 === 0) {
                interval = setInterval(updateTimer, 10);
                timer_button.innerHTML = "STOP";
            } else {
                clearInterval(interval);
                timer_button.innerHTML = "START";
            }
            count++;
        }


        // タイマー更新
        function updateTimer() {
            millis++;
            if (millis === 100) {
                seconds++;
                millis = 0;
            }
            if (seconds === 60) {
                minutes++;
                seconds = 0;
            }
            milli.innerHTML = millis.toString().padStart(2, "0");
            second.innerHTML = seconds.toString().padStart(2, "0") + ":";
            minute.innerHTML = minutes.toString().padStart(2, "0") + ":";
        }

            var gamepads = navigator.getGamepads ? navigator.getGamepads() : (navigator.webkitGetGamepads ? navigator.webkitGetGamepads : []);
            var gamepad = gamepads[0]; // 0番目のゲームパッドを取得

            if (gamepad) {
                // ジョイスティックの値を取得
                lx = gamepad.axes[0] * 100 + 100;
                ly = gamepad.axes[1] * -100 + 100;
                rx = gamepad.axes[2] * 100 + 100;
                ry = gamepad.axes[3] * -100 + 100;

                cross = gamepad.buttons[0].pressed ? 1 : 0;
                circle = gamepad.buttons[1].pressed ? 1 : 0;
                square = gamepad.buttons[2].pressed ? 1 : 0;
                triangle = gamepad.buttons[3].pressed ? 1 : 0;
                l1 = gamepad.buttons[4].pressed ? 1 : 0;
                r1 = gamepad.buttons[5].pressed ? 1 : 0;
                l2 = gamepad.buttons[6].pressed ? 1 : 0; //value * 100;　押し感はこっち
                r2 = gamepad.buttons[7].pressed ? 1 : 0; //value * 100;　押し感はこっち
                create = gamepad.buttons[8].pressed ? 1 : 0;
                option = gamepad.buttons[9].pressed ? 1 : 0;
                l3 = gamepad.buttons[10].pressed ? 1 : 0;
                r3 = gamepad.buttons[11].pressed ? 1 : 0;
                up = gamepad.buttons[12].pressed ? 1 : 0;
                down = gamepad.buttons[13].pressed ? 1 : 0;
                left = gamepad.buttons[14].pressed ? 1 : 0;
                right = gamepad.buttons[15].pressed ? 1 : 0;
                ps = gamepad.buttons[16].pressed ? 1 : 0;
                touchpad = gamepad.buttons[17].pressed ? 1 : 0;
            }
        }

        setInterval(updateGamepadValues, 0);

        //通信
        setInterval(() => {
            behavior_list.innerHTML = behavior;
            Send_Data = `${behavior},${mode},${button[0]},${button[1]},${button[2]},${color},${STOP},${lx.toFixed(0)},${ly.toFixed(0)},${rx.toFixed(0)},${ry.toFixed(0)}`;
            ws.send(Send_Data);
        }, 0);

        ws.onmessage = function (event) {
            var receiveData = event.data;
            var dataArray = receiveData.split(",");
            position_x = parseInt(dataArray[0]);
            position_y = parseInt(dataArray[1]);
            position_direction = parseInt(dataArray[2]);
            if (color == 1) {
                robot.style.right = (position_x * 0.1 + 16.5) + "px";
            } else if (color == 0) {
                robot.style.right = (-position_x * 0.1 + 366.5) + "px";
            }
            robot.style.bottom = (position_y * 0.1 + 16.55) + "px";
            robot.style.transform = "rotate(" + position_direction + "deg)";
            position_X.innerHTML = "X: " + position_x;
            position_Y.innerHTML = "Y: " + position_y;
        }


        // 初期設定関数の呼び出し
        setup();
    </script>
